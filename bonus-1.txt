	public int getlastID() {
		String[] s = getFile("ItemsBought.txt");
		String[] y = s[s.length-1].split(", ");
		int num = Integer.parseInt(y[0].substring(1, y[0].length()));
		num++;
		return num;
	}
	
	public boolean search(String currentUser) {
		String[] h = getFile("ItemsBought.txt");
		for (int i=0;i<h.length;i++) { 
			if (h[i].equals(currentUser)) {
				return true;				
			}
		}
		return false;
	}
	
	public void enterNewCheckout(String confirmID, String username, String productName, int totalCost ){
		boolean flag = false;
		String[] h = getFile("ItemsBought.txt");
		for (int i=0;i<h.length;i++) { 
			if (h[i].equals(username)) {
				flag = true;
			}
			if (h[i] == null && flag){
				h[i] = confirmID+", "+ productName+", " + totalCost + "\n";
				flag = false;
			}
		}
		return;
	}
